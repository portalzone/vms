# ----------------------------------------------------------------------
# STAGE 1: BUILD FRONTEND (Vue/Vite)
# ----------------------------------------------------------------------
FROM node:20 AS frontend_builder

WORKDIR /app

# Copy package files for caching
COPY vms-frontend/vue-project/package*.json ./

# Install dependencies
RUN npm ci

# Copy source code
COPY vms-frontend/vue-project/ ./

# Build assets
ENV ROLLUP_DISABLE_NATIVE=true
RUN npm run build

# ----------------------------------------------------------------------
# STAGE 2: PRODUCTION (Laravel + PHP-FPM + Nginx)
# ----------------------------------------------------------------------
FROM php:8.3-fpm-alpine AS final

# Install system & PHP extensions
RUN apk add --no-cache \
    nginx \
    supervisor \
    libzip-dev \
    libpng-dev \
    libxml2-dev \
    $PHPIZE_DEPS \
    && docker-php-ext-install pdo_mysql opcache zip gd bcmath \
    && apk del $PHPIZE_DEPS

WORKDIR /var/www/html

# Install Composer
COPY --from=composer:latest /usr/bin/composer /usr/bin/composer

# Copy application source first (so artisan exists)
COPY . .

# Install PHP dependencies
RUN composer install --no-dev --optimize-autoloader

# Copy NGINX config
COPY .render/nginx.conf /etc/nginx/conf.d/default.conf

# Copy Supervisor config
COPY .render/supervisord.conf /etc/supervisord.conf

# Copy built Vue assets
COPY --from=frontend_builder /app/dist/ /var/www/html/public/build/

# Set permissions
RUN chown -R www-data:www-data /var/www/html/storage /var/www/html/bootstrap/cache \
    && chmod -R 775 /var/www/html/storage /var/www/html/bootstrap/cache

# Render requires binding to $PORT
EXPOSE ${PORT}

CMD ["/bin/sh", "-c", "supervisord -c /etc/supervisord.conf"]
